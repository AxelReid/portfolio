# Testing

```jsx
import all_links from 'config/links'
import { useRouter } from 'next/router'

const findPathObj = (path, links) => {
  let result = null
  for (const item of links) {
    if (item.link?.pathname === (path?.includes('/') ? path : '/' + path))
      result = item
    if (item.sublinks) result = findPathObj(path, item.sublinks)
    if (result) break
  }
  return result
}

export const findPageName = (path = current_path(), links = all_links) => {
  return findPathObj(path, links)?.title
}

export const findBreadcrumb = (paths, links) =>
  paths.map(
    (path) => findPathObj(path, links) || { title: document.title, link: null }
  )

export const current_path = () => {
  const router = useRouter()
  const splitted = router.pathname
    .split('/')
    .filter(
      (sp) =>
        sp !== '' && sp !== 'en' && sp !== 'ru' && sp !== 'uz' && sp !== 'oz'
    )
  return splitted[splitted.length - 1]
}
```
